function exportInvCSV(invoices) { let csvContent = 'data:text/csv;charset=utf-8,Invoice ID,Product Name,Cost 1,Cost 2\n'; invoices.forEach((p) => { csvContent += `${p.id},${p.productName},${p.costs[0]},${p.costs[1]}\n`; }); const encodedUri = encodeURI(csvContent); const link = document.createElement('a'); link.setAttribute('href', encodedUri); link.setAttribute('target', '_blank'); link.setAttribute('download', 'invoices.csv'); document.body.appendChild(link); link.click(); }
